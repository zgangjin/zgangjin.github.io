Html5学习笔记——canvas贪吃蛇

　先说下大概思路：
贪吃蛇这个游戏大家都并不陌生，这里就直入主题了，首先是方块的大小，和随机生成的方块的位置，必须保证他们会完全重合。这样才好判断相交，看起来更好看，

代码：这里方块的大小为2020；
var bx=Math.floor((Math.random()29)+1)20;
var by=Math.floor(Math.random()14+1)*20;
然后后就是判断相交和位置:
相交这里用的是判断两个方块的坐标完全重合即为相交。
位置，用一个数组记录起蛇头所走过的坐标但不要会记下来设定一个数组的长度限制，每相交一次此限制长度加1
然后蛇身的坐标用此数组里面记的坐标。跟着蛇头一起刷新。
下面为实现核心代码：
var canvas=document.getElementById(‘Mycanvas’);
var cantext=canvas.getContext(‘2d’);
var ax=0;
var ay=0;

//键盘各个方向的速度；
var uv=0;
var dv=0;
var lv=0;    
var rv=0;

var arrcoord=[];

var cx=0;
var cy=0;

//禁止反方向运行；
var xboo=true;
var yboo=true;

var mainlenght=1;
var overboo=true;    

var bx=Math.floor((Math.random()*29)+1)*20;
var by=Math.floor(Math.random()*14+1)*20;    
window.addEventListener('keydown',mdown,true);    
enterframe();
function enterframe(){
    cantext.clearRect(0,0,canvas.width,canvas.height);
    cantext.fillStyle='#999999';
    cantext.fillRect(0,0,canvas.width,canvas.height);
    cantext.fillStyle='#348943';            
    arrcoord.push([ax,ay]);
    if(arrcoord.length>=(mainlenght+1))
    {
        arrcoord.shift();
    }
    drawlenght();
    draw();    
    gamevoer();
    ax+=lv+rv;
    ay+=uv+dv;
    //cantext.fill();
    if(ax==bx && ay==by)
    {
        mainlenght++;
        bx=Math.floor((Math.random()*29)+1)*20;
        by=Math.floor(Math.random()*14+1)*20;

    }

    if(overboo)
    {
        setTimeout(enterframe,200);
    }
}

function draw(){        
    cantext.fillRect(bx,by,20,20);            
}
//键盘控制；
function mdown(e){
    var md=e.keyCode;
    switch(md)
    {
        case 37:        
        if(xboo)
        {    
            uv=0;
            dv=0;
            lv=-20;    
            rv=0;
            xboo=false;
            yboo=true;
        }
        break;
        case 39:
        if(xboo)
        {
            uv=0;
            dv=0;
            lv=0;    
            rv=20;
            xboo=false;
            yboo=true;
        }
        break;
        case 38:
        if(yboo)
        {
            uv=-20;
            dv=0;
            lv=0;    
            rv=0;
            xboo=true;
            yboo=false;
        }
        break;
        case 40:
        if(yboo)
        {
            uv=0;
            dv=20;
            lv=0;    
            rv=0;
            xboo=true;
            yboo=false;    
        }
        break; 
    }

}

function drawlenght(){
        for(var i=0;i<mainlenght;i++)
        {
            cantext.beginPath();

            cantext.fillRect(arrcoord[i][0],arrcoord[i][1],20,20);
            cantext.closePath()
        }
}

function gamevoer(){
    if(arrcoord.length>4)
    {
        for(var i=3;i<arrcoord.length;i++)
        {
            if(arrcoord[0][0]==arrcoord[i][0] && arrcoord[0][1]==arrcoord[i][1])
            {
                overboo=false;
                alert('⊙﹏⊙游戏结束');
            }
        }
    }
    if(arrcoord[0][0]<0 || arrcoord[0][0]> canvas.width || arrcoord[0][1]<0 || arrcoord[0][1]>canvas.height)
    {
            overboo=false;
            alert('⊙﹏⊙游戏结束');
    }
}
